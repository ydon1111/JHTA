aca_ori_df = pd.read_csv('C:/Users/user/Desktop/semi2/aca/ACA_PRI.csv', header=None) #    

big_mac = pd.read_csv('C:/Users/user/Desktop/semi2/aca/big_mac_2019.csv') # 2019 
gdp = pd.read_csv('C:/Users/user/Desktop/semi2/aca/OECD_GDP.csv') # GDP
happiness = pd.read_csv('C:/Users/user/Desktop/semi2/aca//happiness_2019.csv') # , GNI

aca_ori_df = aca_ori_df.drop([0,2], axis=1) #   

aca_ori_df.columns = ['country', 'aca_price'] #  
big_mac.rename(columns = {'price' : 'bm_price'}, inplace = True) #  
gdp.rename(columns = {'Country' : 'country', 'GDP\n(nominal, 2017)' : 'gdp'}, inplace = True) #  
happiness.columns = ['rank', 'country', 'happiness_score', 'gni'] #  

aca_ori_df['aca_price'] = aca_ori_df['aca_price'].str.slice(start = 1).apply(pd.to_numeric) #   
gdp['gdp'] = gdp['gdp'].str.replace(',','').str.slice(start = 1).apply(pd.to_numeric) # GDP  
happiness['gni'] = happiness['gni'].apply(pd.to_numeric) # GNI  

US_aca_price = aca_ori_df[aca_ori_df['country'] == 'United States']['aca_price'].values[0] #   
US_bm_price = big_mac[big_mac['country'] == 'United States']['bm_price'].values[0] #   

#  
aca_ori_df['aca_index'] = (aca_ori_df['aca_price'] / US_aca_price).round(3) #    ( 3)
big_mac['bm_index'] = (big_mac['bm_price'] / US_bm_price).round(3) #    ( 3)

index_df = pd.merge(aca_ori_df, big_mac, on="country", how = "outer") # df df 
index_df = pd.merge(index_df, gdp, on="country", how = "outer") # GDP df 
index_df = pd.merge(index_df, happiness, on="country", how = "left")

index_df = index_df.sort_values(by=['bm_price'], axis=0, ascending=False).dropna(axis=0) # NA , 

index_df = index_df[['country', 'aca_price', 'bm_price', 'aca_index', 'bm_index', 'happiness_score', 'gdp', 'gni']]

#  
index_df = index_df[index_df['country'] != index_df.loc[(index_df['happiness_score'] <= 4.5) & (index_df['aca_index'] >= 0.7)]['country'].values[0]]
