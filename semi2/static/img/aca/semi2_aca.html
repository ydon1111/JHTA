aca_ori_df = pd.read_csv('C:/Users/user/Desktop/semi2/aca/ACA_PRI.csv', header=None) # 세계 아메리카노 가격 가져오기

big_mac = pd.read_csv('C:/Users/user/Desktop/semi2/aca/big_mac_2019.csv') # 2019년 빅맥지수
gdp = pd.read_csv('C:/Users/user/Desktop/semi2/aca/OECD_GDP.csv') # GDP
happiness = pd.read_csv('C:/Users/user/Desktop/semi2/aca//happiness_2019.csv') # 행복지수, GNI

aca_ori_df = aca_ori_df.drop([0,2], axis=1) # 필요없는 컬럼 삭제

aca_ori_df.columns = ['country', 'aca_price'] # 컬럼명 지정
big_mac.rename(columns = {'price' : 'bm_price'}, inplace = True) # 컬럼명 수정
gdp.rename(columns = {'Country' : 'country', 'GDP\n(nominal, 2017)' : 'gdp'}, inplace = True) # 컬럼명 수정
happiness.columns = ['rank', 'country', 'happiness_score', 'gni'] # 컬럼명 지정

aca_ori_df['aca_price'] = aca_ori_df['aca_price'].str.slice(start = 1).apply(pd.to_numeric) # 가격을 숫자형으로 변환
gdp['gdp'] = gdp['gdp'].str.replace(',','').str.slice(start = 1).apply(pd.to_numeric) # GDP를 숫자형으로 변환
happiness['gni'] = happiness['gni'].apply(pd.to_numeric) # GNI를 숫자형으로 변환

US_aca_price = aca_ori_df[aca_ori_df['country'] == 'United States']['aca_price'].values[0] # 미국 아메리카노 가격
US_bm_price = big_mac[big_mac['country'] == 'United States']['bm_price'].values[0] # 미국 빅맥 가격

# 지수 계산
aca_ori_df['aca_index'] = (aca_ori_df['aca_price'] / US_aca_price).round(3) # 아카 지수 계산 (소수점 3자리)
big_mac['bm_index'] = (big_mac['bm_price'] / US_bm_price).round(3) # 빅맥 지수 계산 (소수점 3자리)

index_df = pd.merge(aca_ori_df, big_mac, on="country", how = "outer") # 아카df와 빅맥df 합치기
index_df = pd.merge(index_df, gdp, on="country", how = "outer") # GDP와 지수df 합치기
index_df = pd.merge(index_df, happiness, on="country", how = "left")

index_df = index_df.sort_values(by=['bm_price'], axis=0, ascending=False).dropna(axis=0) # NA값 삭제, 정렬

index_df = index_df[['country', 'aca_price', 'bm_price', 'aca_index', 'bm_index', 'happiness_score', 'gdp', 'gni']]

# 이상치 제거
index_df = index_df[index_df['country'] != index_df.loc[(index_df['happiness_score'] <= 4.5) & (index_df['aca_index'] >= 0.7)]['country'].values[0]]
